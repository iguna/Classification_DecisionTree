In [11]:
#initialize and fit the gini metric-based DT classifier
clf_gini = DecisionTreeClassifier()
clf_gini.fit(x,y)

#viz code
dot_data = StringIO()

#create DT based on gini metric
export_graphviz(
                clf_gini,
                out_file=dot_data,
                feature_names = x.columns,
                class_names = y.columns[0],
                rounded = True,
                filled = True
                )
graph = pydotplus.graph_from_dot_data(dot_data.getvalue())
#print gini metric-based DT as png image
Image(graph.create_png())
